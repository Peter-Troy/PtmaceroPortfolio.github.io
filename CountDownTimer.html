Input Hours, Minutes, and Seconds in the input field:

Click the "Start Countdown" Button

Time will start to countdown.

Javascript Code:

								
// Wait for the DOM to be fully loaded before executing the script
document.addEventListener("DOMContentLoaded", function() {
// Get references to the start button and timer display elements
const startBtn = document.getElementById("startBtn");
const timerDisplay = document.getElementById("timer");

// Attach a click event listener to the start button
startBtn.addEventListener("click", startCountdown);

// Function to handle the countdown when the start button is clicked
function startCountdown() {
// Retrieve user input for hours, minutes, and seconds
const hours = parseInt(document.getElementById("hours").value) || 0;
const minutes = parseInt(document.getElementById("minutes").value) || 0;
const seconds = parseInt(document.getElementById("seconds").value) || 0;

// Calculate the total time in seconds
const totalSeconds = hours * 3600 + minutes * 60 + seconds;
let remainingTime = totalSeconds;

// Function to update the timer display
function updateTimer() {
// Calculate hours, minutes, and seconds remaining
const hoursRemaining = Math.floor(remainingTime / 3600);
const minutesRemaining = Math.floor((remainingTime % 3600) / 60);
const secondsRemaining = remainingTime % 60;

// Update the timer display with the formatted time
timerDisplay.textContent = `${hoursRemaining.toString().padStart(2, '0')} : ${minutesRemaining.toString().padStart(2, '0')} : ${secondsRemaining.toString().padStart(2, '0')}`;

// Continue updating the timer if time remains, otherwise display "Countdown Complete!"
if (remainingTime > 0) {
remainingTime--;
setTimeout(updateTimer, 1000);
} else {
timerDisplay.textContent = "Countdown Complete!";
}
}

// Initial call to start the countdown
updateTimer();
}
});
